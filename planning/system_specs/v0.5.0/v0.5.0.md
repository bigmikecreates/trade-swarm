# v0.5.0 — Infrastructure Hardening

## Goal
Stop running the system as a script on your laptop. Move everything into Docker containers, replace SQLite with Postgres, add Grafana monitoring, wire the kill switch to Telegram, and build a full audit log. The system must be able to run unattended for 48 hours without human intervention and without data loss.

## Gate to enter this version
- 8 weeks paper trading, Sharpe > 1.0, drawdown < 15% ✓
- All 3 agents active and orchestrator wiring confirmed ✓
- Redis pub/sub stable ✓

## Gate to exit this version
- System runs 48 hours unattended on a VPS with no crashes
- No data loss if any single container restarts
- Kill switch reachable and tested from Telegram
- Grafana dashboard showing live P&L, positions, agent status
- All gate rules from the full 16-rule risk gate implemented

### 2. Postgres schema (replaces SQLite)

```sql
-- sql/init.sql

CREATE TABLE IF NOT EXISTS trades (
    id              SERIAL PRIMARY KEY,
    timestamp       TIMESTAMPTZ DEFAULT NOW(),
    asset           TEXT NOT NULL,
    direction       TEXT NOT NULL,
    action          TEXT NOT NULL,          -- enter, exit, scale_in, scale_out
    size            NUMERIC(18, 8),
    entry_price     NUMERIC(18, 8),
    exit_price      NUMERIC(18, 8),
    pnl             NUMERIC(18, 8),
    stop_loss       NUMERIC(18, 8),
    order_id        TEXT,
    broker          TEXT,
    status          TEXT,
    slippage_bps    NUMERIC(10, 4),
    latency_ms      NUMERIC(10, 2),
    indicators      JSONB,
    regime          TEXT,
    conviction      NUMERIC(6, 4),
    agent_ids       TEXT[]
);

CREATE TABLE IF NOT EXISTS gate_log (
    id          SERIAL PRIMARY KEY,
    timestamp   TIMESTAMPTZ DEFAULT NOW(),
    result      TEXT NOT NULL,
    rule_fired  TEXT,
    reason      TEXT,
    order_data  JSONB,
    asset       TEXT
);

CREATE TABLE IF NOT EXISTS regime_log (
    id          SERIAL PRIMARY KEY,
    timestamp   TIMESTAMPTZ DEFAULT NOW(),
    asset       TEXT NOT NULL,
    regime      TEXT NOT NULL,
    confidence  NUMERIC(6, 4),
    adx         NUMERIC(8, 4),
    atr_ratio   NUMERIC(8, 4)
);

CREATE TABLE IF NOT EXISTS equity_curve (
    id          SERIAL PRIMARY KEY,
    timestamp   TIMESTAMPTZ DEFAULT NOW(),
    equity      NUMERIC(18, 2),
    daily_pnl   NUMERIC(18, 2),
    drawdown    NUMERIC(6, 4),
    open_positions INTEGER
);

CREATE INDEX idx_trades_asset     ON trades(asset);
CREATE INDEX idx_trades_timestamp ON trades(timestamp);
CREATE INDEX idx_gate_timestamp   ON gate_log(timestamp);
```

import os
import psycopg2
from psycopg2.extras import RealDictCursor, Json
from datetime import datetime

POSTGRES_URL = os.environ["POSTGRES_URL"]

def get_conn():
    return psycopg2.connect(POSTGRES_URL)

def log_trade(trade: dict):
    with get_conn() as conn:
        with conn.cursor() as cur:
            cur.execute("""
                INSERT INTO trades
                (asset, direction, action, size, entry_price, exit_price, pnl,
                 stop_loss, order_id, broker, status, slippage_bps, latency_ms,
                 indicators, regime, conviction)
                VALUES (%s,%s,%s,%s,%s,%s,%s,%s,%s,%s,%s,%s,%s,%s,%s,%s)
            """, (
                trade.get("asset"), trade.get("direction"), trade.get("action"),
                trade.get("size"), trade.get("entry_price"), trade.get("exit_price"),
                trade.get("pnl"), trade.get("stop_loss"), trade.get("order_id"),

### 3. Telegram kill switch + alerting

```python
# alerts/telegram_bot.py
import os, redis
from telegram import Update
from telegram.ext import Application, CommandHandler, ContextTypes

TELEGRAM_TOKEN   = os.environ["TELEGRAM_TOKEN"]
TELEGRAM_CHAT_ID = int(os.environ["TELEGRAM_CHAT_ID"])

r = redis.Redis(decode_responses=True)

async def halt_command(update: Update, context: ContextTypes.DEFAULT_TYPE):
    """Activate kill switch via /halt command."""
    if update.effective_chat.id != TELEGRAM_CHAT_ID:
        return  # ignore unauthorised users
    r.set("kill_switch", "1")
    r.publish("system_halt", '{"reason": "Telegram /halt command"}')
    await update.message.reply_text("⛔ Kill switch activated. All trading halted.")

async def resume_command(update: Update, context: ContextTypes.DEFAULT_TYPE):
    if update.effective_chat.id != TELEGRAM_CHAT_ID:
        return
    r.delete("kill_switch")
    await update.message.reply_text("✅ Kill switch cleared. System can resume.")

async def status_command(update: Update, context: ContextTypes.DEFAULT_TYPE):
    if update.effective_chat.id != TELEGRAM_CHAT_ID:
        return
    equity    = r.get("current_equity") or "unknown"
    drawdown  = r.get("current_drawdown") or "unknown"
    positions = r.smembers("open_positions")
    kill      = r.get("kill_switch") == "1"

    msg = (
        f"{'⛔ HALTED' if kill else '✅ RUNNING'}\n"
        f"Equity: ${equity}\n"
        f"Drawdown: {drawdown}\n"
        f"Open positions: {', '.join(positions) or 'none'}"
    )
    await update.message.reply_text(msg)

def start_bot():
    app = Application.builder().token(TELEGRAM_TOKEN).build()
    app.add_handler(CommandHandler("halt",   halt_command))
    app.add_handler(CommandHandler("resume", resume_command))
    app.add_handler(CommandHandler("status", status_command))
    app.run_polling()

# Proactive alerts — call these from the trading loop
def send_alert(message: str):
    import requests
    url     = f"https://api.telegram.org/bot{TELEGRAM_TOKEN}/sendMessage"
    payload = {"chat_id": TELEGRAM_CHAT_ID, "text": message, "parse_mode": "HTML"}
    try:
        requests.post(url, json=payload, timeout=5)
    except Exception:
        pass   # alerts must never crash the main loop

# send_alert("✅ Trade opened: LONG AAPL 15 shares @ $185.40")
# send_alert("⛔ Gate fired: daily_loss_limit — system halted")
```
