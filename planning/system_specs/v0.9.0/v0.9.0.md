### 2. XGBoost signal confidence layer

The ML layer doesn't replace your rule-based agents. It sits on top and learns which combinations of indicators historically predicted profitable outcomes. Its output replaces `SignalEvent.confidence`.

#### Feature engineering

```python
# ml/features.py
import pandas as pd
import pandas_ta as ta
import numpy as np

def build_features(df: pd.DataFrame) -> pd.DataFrame:
    """
    Build ML features from OHLCV data.
    All features must be calculable at bar-close time only.
    Use .shift(1) on any feature derived from current bar to prevent lookahead.
    """
    df = df.copy()

    # Price-based
    df.ta.rsi(length=14, append=True)
    df.ta.rsi(length=7, append=True)
    df.ta.macd(append=True)
    df.ta.adx(length=14, append=True)
    df.ta.bbands(length=20, append=True)
    df.ta.atr(length=14, append=True)
    df.ta.ema(length=20, append=True)
    df.ta.ema(length=50, append=True)
    df.ta.ema(length=200, append=True)

    # Derived features
    df["ema_cross"]      = (df["EMA_20"] > df["EMA_50"]).astype(int)
    df["price_vs_ema200"] = df["Close"] / df["EMA_200"] - 1
    df["atr_pct"]        = df["ATRr_14"] / df["Close"]
    df["rsi_diff"]       = df["RSI_14"] - df["RSI_7"]

    # Rolling statistics
    for window in [5, 10, 20]:
        df[f"return_{window}"] = df["Close"].pct_change(window).shift(1)
        df[f"vol_{window}"]    = df["Close"].pct_change().rolling(window).std().shift(1)

    # Volume features
    df["volume_ratio"]  = df["Volume"] / df["Volume"].rolling(20).mean()
    df["obv"]           = ta.obv(df["Close"], df["Volume"])

    df.dropna(inplace=True)
    return df

FEATURE_COLUMNS = [
    "RSI_14", "RSI_7", "rsi_diff", "ADX_14", "DMP_14", "DMN_14",
    "MACD_12_26_9", "MACDh_12_26_9",
    "BBP_20_2.0",   # Bollinger band percent
    "ema_cross", "price_vs_ema200", "atr_pct",
    "return_5", "return_10", "return_20",
    "vol_5", "vol_10", "vol_20",
    "volume_ratio",
]
```

#### Label generation

```python
# ml/labels.py
import pandas as pd

def generate_labels(
    df: pd.DataFrame,
    forward_bars: int  = 12,    # predict 12 bars ahead (e.g. 12 hours on 1H data)
    min_return: float  = 0.005, # 0.5% minimum to be a "positive" trade
) -> pd.Series:
    """
    Binary label: 1 if price rises by min_return within forward_bars,
                  0 otherwise.
    CRITICAL: shift forward returns backward so they align with the current bar
    but represent future outcomes.
    """
    future_return = df["Close"].shift(-forward_bars) / df["Close"] - 1
    return (future_return > min_return).astype(int)
```

#### Training

```python
# ml/train.py
import xgboost as xgb
import pandas as pd
import numpy as np
from sklearn.model_selection import TimeSeriesSplit
from sklearn.metrics import roc_auc_score
import joblib
from ml.features import build_features, FEATURE_COLUMNS
from ml.labels import generate_labels
from data.fetcher import fetch_ohlcv

def train_confidence_model(symbol: str, save_path: str) -> xgb.XGBClassifier:
    # Fetch 3+ years of training data
    df      = fetch_ohlcv(symbol, period="3y", interval="1h")
    df      = build_features(df)
    labels  = generate_labels(df)

    # Align features and labels (drop rows where label is NaN)
    df      = df.iloc[:len(labels)]
    df      = df[df.index.isin(labels.index)]
    labels  = labels[labels.index.isin(df.index)]

    X = df[FEATURE_COLUMNS]
    y = labels

    # Time-series cross validation — no shuffling
    tscv    = TimeSeriesSplit(n_splits=5)
    scores  = []

    for train_idx, val_idx in tscv.split(X):
        X_train, X_val = X.iloc[train_idx], X.iloc[val_idx]
        y_train, y_val = y.iloc[train_idx], y.iloc[val_idx]

        model = xgb.XGBClassifier(
            n_estimators     = 300,
            max_depth        = 4,
            learning_rate    = 0.05,
            subsample        = 0.8,
            colsample_bytree = 0.8,
            use_label_encoder = False,
            eval_metric      = "auc",
            random_state     = 42,
        )
        model.fit(X_train, y_train,
                  eval_set=[(X_val, y_val)],
                  verbose=False)

        score = roc_auc_score(y_val, model.predict_proba(X_val)[:, 1])
        scores.append(score)
        print(f"Fold AUC: {score:.4f}")

    print(f"Mean AUC: {np.mean(scores):.4f} ± {np.std(scores):.4f}")

    # Train final model on all data
    final_model = xgb.XGBClassifier(
        n_estimators=300, max_depth=4, learning_rate=0.05,
        subsample=0.8, colsample_bytree=0.8,
        use_label_encoder=False, eval_metric="auc", random_state=42,
    )
    final_model.fit(X, y)
    joblib.dump(final_model, save_path)
    print(f"Model saved to {save_path}")
    return final_model
```

#### Inference — wire into signal agents

```python
# ml/predictor.py
import joblib
import pandas as pd
from ml.features import build_features, FEATURE_COLUMNS

class MLConfidencePredictor:
    def __init__(self, model_path: str):
        self.model = joblib.load(model_path)

    def predict_confidence(self, df: pd.DataFrame) -> float:
        """
        Returns probability 0.0–1.0 that the current setup leads to a profitable trade.
        Replaces the rule-based confidence in SignalEvent.
        """
        features = build_features(df.tail(50))   # only need recent bars for features
        if features.empty:
            return 0.5
        X     = features[FEATURE_COLUMNS].iloc[[-1]]   # latest bar only
        proba = self.model.predict_proba(X)[0][1]
        return round(float(proba), 4)
```

### 3. Walk-forward optimisation

Walk-forward optimisation validates that your strategy parameters (EMA lengths, ADX thresholds, etc.) generalise to unseen data rather than being overfit to historical data.

```python
# backtest/walk_forward.py
import pandas as pd
import vectorbt as vbt
from data.fetcher import fetch_ohlcv

def walk_forward_test(
    symbol:        str,
    in_sample_bars:  int  = 1000,   # train on 1000 bars
    out_sample_bars: int  = 250,    # test on 250 bars
    n_windows:       int  = 8,
) -> pd.DataFrame:
    """
    Slides a window through history.
    In each window: optimise on in-sample, test on out-of-sample.
    If out-of-sample Sharpe decays significantly → strategy is overfit.
    """
    df      = fetch_ohlcv(symbol, period="5y", interval="1h")
    results = []

    for i in range(n_windows):
        start       = i * out_sample_bars
        in_end      = start + in_sample_bars
        out_end     = in_end + out_sample_bars

        if out_end > len(df):
            break

        in_sample   = df.iloc[start:in_end]
        out_sample  = df.iloc[in_end:out_end]

        # Optimise EMA lengths on in-sample
        best_sharpe  = -np.inf
        best_params  = {}
        for fast in [10, 15, 20, 25]:
            for slow in [40, 50, 60, 75]:
                if fast >= slow:
                    continue
                sharpe = _backtest_ema_cross(in_sample, fast, slow)
                if sharpe > best_sharpe:
                    best_sharpe = sharpe
                    best_params = {"fast": fast, "slow": slow}

        # Test best params on out-of-sample
        oos_sharpe = _backtest_ema_cross(out_sample, **best_params)
        results.append({
            "window":       i + 1,
            "in_sharpe":    round(best_sharpe, 3),
            "out_sharpe":   round(oos_sharpe, 3),
            "params":       best_params,
            "decay":        round(best_sharpe - oos_sharpe, 3),
        })
        print(f"Window {i+1}: IS={best_sharpe:.2f} OOS={oos_sharpe:.2f} params={best_params}")

    return pd.DataFrame(results)

def _backtest_ema_cross(df: pd.DataFrame, fast: int, slow: int) -> float:
    import pandas_ta as ta
    df  = df.copy()
    df.ta.ema(length=fast, append=True)
    df.ta.ema(length=slow, append=True)
    df.dropna(inplace=True)
    entries = df[f"EMA_{fast}"] > df[f"EMA_{slow}"]
    exits   = df[f"EMA_{fast}"] < df[f"EMA_{slow}"]
    pf      = vbt.Portfolio.from_signals(df["Close"], entries, exits,
                                         freq="1h", fees=0.001)
    return float(pf.sharpe_ratio())
```

**Passing criterion:** Average out-of-sample Sharpe > 0.7, decay (IS minus OOS) < 0.5 across all windows. If decay is large, your parameters are overfit and need to be simplified.

### 5. Agent weight tracker (deferred from v0.4.0)

Now that you have hundreds of closed trades, enable the learning weight system in the orchestrator:

```python
# agents/orchestrator/weight_tracker.py
import redis

class AgentWeightTracker:
    def __init__(self, learning_rate: float = 0.05):
        self.lr = learning_rate
        self.r  = redis.Redis(decode_responses=True)

    def get_weight(self, agent_id: str) -> float:
        val = self.r.get(f"agent_weight:{agent_id}")
        return float(val) if val else 0.5

    def update(self, agent_id: str, trade_was_profitable: bool):
        current = self.get_weight(agent_id)
        reward  = 1.0 if trade_was_profitable else 0.0
        new_w   = current + self.lr * (reward - current)
        new_w   = max(0.1, min(0.95, new_w))
        self.r.set(f"agent_weight:{agent_id}", new_w)
        return new_w
```
