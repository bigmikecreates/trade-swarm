# v0.4.0 — Orchestrator + Risk Agent + Sentiment

## Goal
Wire all specialist agents through a LangGraph orchestrator that makes final trade decisions. Add a full `RiskAgent` with VaR and drawdown tracking. Add a `SentimentAgent` using FinBERT on Alpaca news. Expand to 2–3 assets.

## Gate to enter this version
- Both `TrendSignalAgent` and `MeanReversionAgent` profitable independently ✓
- Regime detection confirmed switching strategies correctly ✓
- Redis pub/sub wiring stable with no missed messages ✓

## Gate to exit this version
- 8 weeks continuous paper trading with all agents active
- Portfolio Sharpe > 1.0 across the full 8-week period
- Max drawdown < 15% at any point
- All three assets trading independently with correct position sizing
- Orchestrator resolving conflicts correctly (visible in logs)

### 1. SentimentAgent

```python
# agents/sentiment/sentiment_agent.py
import time, threading, json
from dataclasses import dataclass
from datetime import datetime
from collections import deque
import math, redis
import alpaca_trade_api as tradeapi
from transformers import pipeline
from sentence_transformers import SentenceTransformer
import numpy as np

@dataclass
class SentimentEvent:
    asset:         str
    score:         float    # -1.0 → +1.0
    magnitude:     float    # 0.0 → 1.0
    source_type:   str
    source_weight: float
    headline:      str
    timestamp:     datetime

class SentimentAgent:
    SOURCE_WEIGHTS = {
        "news": 0.85,
        "social": 0.30,
    }

    def __init__(self, watchlist: list[str], alpaca_key: str, alpaca_secret: str):
        self.watchlist  = watchlist
        self.api        = tradeapi.REST(alpaca_key, alpaca_secret,
                                        "https://paper-api.alpaca.markets")
        self.finbert    = pipeline("text-classification",
                                   model="ProsusAI/finbert",
                                   return_all_scores=True)
        self.dedup_model = SentenceTransformer("all-MiniLM-L6-v2")
        self.r           = redis.Redis(decode_responses=False)
        self.windows: dict[str, "SentimentWindow"] = {
            asset: SentimentWindow(half_life_seconds=3600)
            for asset in watchlist
        }

    def score_headline(self, text: str) -> dict:
        scores    = self.finbert(text[:512])[0]
        score_map = {s["label"]: s["score"] for s in scores}
        return {
            "score":     round(score_map["positive"] - score_map["negative"], 4),
            "magnitude": round(max(score_map["positive"], score_map["negative"]), 4),
        }

    def is_duplicate(self, headline: str, threshold: float = 0.92) -> bool:
        new_emb = self.dedup_model.encode(headline)
        cached  = self.r.lrange("sentiment_dedup", 0, 99)
        for raw in cached:
            old_emb = np.frombuffer(raw, dtype=np.float32)
            sim     = np.dot(new_emb, old_emb) / (
                      np.linalg.norm(new_emb) * np.linalg.norm(old_emb) + 1e-9)
            if sim > threshold:
                return True
        self.r.lpush("sentiment_dedup", new_emb.astype(np.float32).tobytes())
        self.r.expire("sentiment_dedup", 7200)
        return False

    def relevant_assets(self, headline: str) -> list[str]:
        headline_lower = headline.lower()
        return [a for a in self.watchlist if a.lower() in headline_lower]

    def poll_and_score(self):
        """Fetch latest news, score, update windows, publish."""
        try:
            news = self.api.get_news(symbols=self.watchlist, limit=50)
        except Exception as e:
            print(f"[Sentiment] News fetch error: {e}")
            return

        for article in news:
            headline = article.headline
            if self.is_duplicate(headline):
                continue

            scores  = self.score_headline(headline)
            assets  = self.relevant_assets(headline)

            for asset in assets:
                self.windows[asset].add(
                    score         = scores["score"],
                    magnitude     = scores["magnitude"],
                    source_weight = self.SOURCE_WEIGHTS["news"],
                )
                agg = self.windows[asset].aggregate()
                event = SentimentEvent(
                    asset        = asset,
                    score        = round(agg, 4),
                    magnitude    = scores["magnitude"],
                    source_type  = "news",
                    source_weight = self.SOURCE_WEIGHTS["news"],
                    headline     = headline,
                    timestamp    = datetime.utcnow(),
                )
                r = redis.Redis(decode_responses=True)
                r.set(f"sentiment:{asset}", json.dumps({
                    "score": event.score,
                    "magnitude": event.magnitude,
                    "timestamp": event.timestamp.isoformat(),
                }))

    def start(self, interval_seconds: int = 300):
        def loop():
            while True:
                self.poll_and_score()
                time.sleep(interval_seconds)
        threading.Thread(target=loop, daemon=True).start()


class SentimentWindow:
    def __init__(self, half_life_seconds: float):
        self.half_life = half_life_seconds
        self.events: deque = deque(maxlen=500)

    def add(self, score: float, magnitude: float, source_weight: float):
        self.events.append({
            "score": score, "magnitude": magnitude,
            "weight": source_weight, "ts": time.time()
        })

    def aggregate(self) -> float:
        now = time.time()
        weighted_sum, total_weight = 0.0, 0.0
        for e in self.events:
            decay = math.exp(-(now - e["ts"]) * math.log(2) / self.half_life)
            w     = decay * e["magnitude"] * e["weight"]
            weighted_sum += e["score"] * w
            total_weight += w
        return weighted_sum / total_weight if total_weight > 0 else 0.0
```

### 2. Full RiskAgent

```python
# risk/risk_agent.py
import numpy as np
import pandas as pd
from dataclasses import dataclass
from datetime import datetime
from enum import Enum

class RiskDecision(Enum):
    APPROVE = "approve"
    RESIZE  = "resize"
    REJECT  = "reject"
    HALT    = "halt"

@dataclass
class RiskEvent:
    decision:         RiskDecision
    approved_size:    float
    rejection_reason: str | None
    portfolio_heat:   float
    current_drawdown: float
    daily_pnl:        float
    timestamp:        datetime

class RiskAgent:
    def __init__(self, account_equity: float):
        self.equity        = account_equity
        self.peak_equity   = account_equity
        self.daily_start   = account_equity

    def evaluate(self, proposal: dict, market: dict) -> RiskEvent:
        equity  = market.get("account_equity", self.equity)
        size    = proposal.get("suggested_size", 0)
        price   = market.get(f"price:{proposal['asset']}", 0)
        notional = size * price

        # Drawdown check
        self.peak_equity   = max(self.peak_equity, equity)
        drawdown           = (self.peak_equity - equity) / self.peak_equity

        if drawdown >= 0.15:
            return RiskEvent(RiskDecision.HALT, 0.0,
                f"Drawdown {drawdown:.1%} hit 15% limit",
                1.0, drawdown, equity - self.daily_start, datetime.utcnow())

        # Position sizing
        max_notional = equity * 0.10   # 10% max per position
        if notional > max_notional:
            approved_size = max_notional / price if price > 0 else 0
            return RiskEvent(RiskDecision.RESIZE, approved_size,
                f"Resized from {size:.2f} to {approved_size:.2f}",
                notional / equity, drawdown,
                equity - self.daily_start, datetime.utcnow())

        # VaR check
        returns  = pd.Series(market.get("portfolio_returns", []))
        if len(returns) >= 20:
            var_95   = abs(np.percentile(returns, 5))
            if var_95 > 0.02:   # 2% daily VaR limit
                return RiskEvent(RiskDecision.REJECT, 0.0,
                    f"Portfolio VaR {var_95:.1%} exceeds 2% limit",
                    1.0, drawdown, equity - self.daily_start, datetime.utcnow())

        heat = min(notional / equity, 1.0)
        return RiskEvent(RiskDecision.APPROVE, size, None, heat,
            drawdown, equity - self.daily_start, datetime.utcnow())
```

### 3. LangGraph Orchestrator

```python
# agents/orchestrator/orchestrator.py
from langgraph.graph import StateGraph, END
from typing import TypedDict
import redis, json

class OrchestratorState(TypedDict):
    asset:           str
    signal_events:   list
    sentiment_score: float
    regime:          dict
    risk_event:      dict | None
    trade_proposal:  dict | None
    decision:        str
    portfolio_state: dict

r = redis.Redis(decode_responses=True)

def collect_signals(state: OrchestratorState) -> OrchestratorState:
    raw     = r.lrange(f"signals:{state['asset']}", 0, 9)
    signals = [json.loads(s) for s in raw]
    return {**state, "signal_events": signals}

def apply_regime_filter(state: OrchestratorState) -> OrchestratorState:
    active  = state.get("regime", {}).get("active_strategies", [])
    filtered = [s for s in state["signal_events"]
                if s.get("agent_id") in active]
    return {**state, "signal_events": filtered}

def compute_conviction(state: OrchestratorState) -> OrchestratorState:
    signals = state["signal_events"]
    if not signals:
        return {**state, "decision": "abstain", "trade_proposal": None}

    # Weighted vote
    votes = {"long": 0.0, "short": 0.0, "flat": 0.0}
    total = 0.0
    for s in signals:
        w = s["strength"] * s["confidence"]
        votes[s["direction"]] += w
        total += w

    if total == 0:
        return {**state, "decision": "abstain", "trade_proposal": None}

    for k in votes:
        votes[k] /= total

    winner    = max(votes, key=votes.get)
    agreement = votes[winner]

    # Sentiment modifier
    sentiment = state.get("sentiment_score", 0.0)
    signal_dir = 1 if winner == "long" else -1
    sentiment_boost = sentiment * signal_dir * 0.15

    conviction = min(1.0, max(0.0, agreement + sentiment_boost))
    regime_mult = state.get("regime", {}).get("size_multiplier", 1.0)

    if winner == "flat" or conviction < 0.50:
        return {**state, "decision": "abstain", "trade_proposal": None}

    proposal = {
        "asset":          state["asset"],
        "action":         "enter",
        "direction":      winner,
        "conviction":     round(conviction, 4),
        "suggested_size": _base_size(state["portfolio_state"], regime_mult),
        "urgency":        "normal",
        "rationale": {
            "agreement":   agreement,
            "sentiment":   sentiment,
            "regime":      state.get("regime", {}).get("regime"),
        }
    }
    return {**state, "decision": "trade", "trade_proposal": proposal}

def send_to_risk(state: OrchestratorState) -> OrchestratorState:
    r.publish("risk_requests", json.dumps(state["trade_proposal"]))
    return state

def route(state: OrchestratorState) -> str:
    return state["decision"]

def _base_size(portfolio_state: dict, regime_mult: float) -> float:
    equity = portfolio_state.get("equity", 10000)
    return round((equity * 0.02 * regime_mult), 2)   # 2% base risk

# Build graph
workflow = StateGraph(OrchestratorState)
workflow.add_node("collect_signals",    collect_signals)
workflow.add_node("apply_regime_filter", apply_regime_filter)
workflow.add_node("compute_conviction", compute_conviction)
workflow.add_node("send_to_risk",       send_to_risk)

workflow.set_entry_point("collect_signals")
workflow.add_edge("collect_signals",      "apply_regime_filter")
workflow.add_edge("apply_regime_filter",  "compute_conviction")
workflow.add_conditional_edges("compute_conviction", route, {
    "trade":   "send_to_risk",
    "abstain": END,
})
workflow.add_edge("send_to_risk", END)

orchestrator_graph = workflow.compile()
```
