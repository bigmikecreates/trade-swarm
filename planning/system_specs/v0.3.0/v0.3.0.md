## What to build

### 1. MeanReversionAgent

Designed for ranging markets. Enters when price is statistically stretched from its mean and likely to snap back.

```python
# agents/signal/mean_reversion_agent.py
import pandas as pd
import pandas_ta as ta
from dataclasses import dataclass
from datetime import datetime
from agents.signal.trend_agent import Direction, SignalEvent

class MeanReversionAgent:
    def __init__(self, symbol: str):
        self.symbol      = symbol
        self.agent_id    = "MeanReversionAgent"
        self.active      = False   # controlled by regime agent
        self.size_mult   = 1.0

    def generate(self, df: pd.DataFrame) -> SignalEvent | None:
        if not self.active:
            return None

        df = df.copy()
        df.ta.rsi(length=14, append=True)
        df.ta.bbands(length=20, std=2, append=True)
        df.ta.atr(length=14, append=True)
        df.dropna(inplace=True)

        last    = df.iloc[-1]
        close   = last["Close"]
        rsi     = last["RSI_14"]
        bb_low  = last["BBL_20_2.0"]
        bb_high = last["BBU_20_2.0"]
        bb_mid  = last["BBM_20_2.0"]
        atr     = last["ATRr_14"]

        # Z-score: how many std devs from the 20-period mean?
        mean    = df["Close"].rolling(20).mean().iloc[-1]
        std     = df["Close"].rolling(20).std().iloc[-1]
        z_score = (close - mean) / std if std > 0 else 0

        # Long signal: oversold, touching lower band, RSI < 35
        if close <= bb_low and rsi < 35 and z_score < -1.5:
            direction  = Direction.LONG
            strength   = min(abs(z_score) / 3, 1.0)
            confidence = min((35 - rsi) / 35, 1.0)

        # Short signal: overbought, touching upper band, RSI > 65
        elif close >= bb_high and rsi > 65 and z_score > 1.5:
            direction  = Direction.SHORT
            strength   = min(abs(z_score) / 3, 1.0)
            confidence = min((rsi - 65) / 35, 1.0)

        else:
            direction  = Direction.FLAT
            strength   = 0.0
            confidence = 0.0

        return SignalEvent(
            agent_id   = self.agent_id,
            asset      = self.symbol,
            direction  = direction,
            strength   = round(strength * self.size_mult, 4),
            confidence = round(confidence, 4),
            timeframe  = "1H",
            timestamp  = datetime.utcnow(),
            expiry     = datetime.utcnow(),
            indicators = {
                "rsi": rsi, "z_score": z_score,
                "bb_low": bb_low, "bb_high": bb_high, "atr": atr
            }
        )
```

### 2. RegimeAgent (rule-based classifier)

```python
# agents/regime/regime_agent.py
import pandas as pd
import pandas_ta as ta
from dataclasses import dataclass
from datetime import datetime
from collections import deque
from enum import Enum

class RegimeType(Enum):
    TRENDING_UP   = "trending_up"
    TRENDING_DOWN = "trending_down"
    RANGING       = "ranging"
    HIGH_VOL      = "high_volatility"
    LOW_VOL       = "low_volatility"

class VolatilityLevel(Enum):
    LOW     = "low"
    NORMAL  = "normal"
    HIGH    = "high"
    EXTREME = "extreme"

@dataclass
class RegimeEvent:
    asset:             str
    regime:            RegimeType
    confidence:        float
    volatility_level:  VolatilityLevel
    active_strategies: list[str]
    size_multiplier:   float
    adx:               float
    atr_ratio:         float
    timestamp:         datetime

class RegimeAgent:
    # Which strategies to activate per regime
    STRATEGY_MAP = {
        RegimeType.TRENDING_UP:   ["TrendSignalAgent"],
        RegimeType.TRENDING_DOWN: ["TrendSignalAgent"],
        RegimeType.RANGING:       ["MeanReversionAgent"],
        RegimeType.HIGH_VOL:      [],                    # nothing trades in chaos
        RegimeType.LOW_VOL:       ["MeanReversionAgent"],
    }
    SIZE_MAP = {
        RegimeType.TRENDING_UP:   1.0,
        RegimeType.TRENDING_DOWN: 1.0,
        RegimeType.RANGING:       0.8,
        RegimeType.HIGH_VOL:      0.25,
        RegimeType.LOW_VOL:       0.75,
    }

    def __init__(self, symbol: str, confirmation_bars: int = 3):
        self.symbol           = symbol
        self.confirmation_bars = confirmation_bars
        self._buffer          = deque(maxlen=confirmation_bars)
        self.current_regime   = None

    def classify(self, df: pd.DataFrame) -> RegimeEvent:
        df = df.copy()
        df.ta.adx(length=14, append=True)
        df.ta.ema(length=20, append=True)
        df.ta.ema(length=50, append=True)
        df.ta.atr(length=14, append=True)
        df.dropna(inplace=True)

        last      = df.iloc[-1]
        adx       = last["ADX_14"]
        dmp       = last["DMP_14"]
        dmn       = last["DMN_14"]
        ema20     = last["EMA_20"]
        ema50     = last["EMA_50"]
        atr_now   = last["ATRr_14"]
        atr_avg   = df["ATRr_14"].rolling(20).mean().iloc[-1]
        atr_ratio = atr_now / atr_avg if atr_avg > 0 else 1.0

        # Volatility level
        if atr_ratio > 2.0:
            vol = VolatilityLevel.EXTREME
        elif atr_ratio > 1.4:
            vol = VolatilityLevel.HIGH
        elif atr_ratio < 0.7:
            vol = VolatilityLevel.LOW
        else:
            vol = VolatilityLevel.NORMAL

        # Regime
        if vol in (VolatilityLevel.HIGH, VolatilityLevel.EXTREME):
            regime, confidence = RegimeType.HIGH_VOL, 0.9
        elif adx > 25 and dmp > dmn and ema20 > ema50:
            regime, confidence = RegimeType.TRENDING_UP, min(adx / 50, 1.0)
        elif adx > 25 and dmn > dmp and ema20 < ema50:
            regime, confidence = RegimeType.TRENDING_DOWN, min(adx / 50, 1.0)
        elif vol == VolatilityLevel.LOW:
            regime, confidence = RegimeType.LOW_VOL, 0.7
        else:
            regime, confidence = RegimeType.RANGING, max(0.0, 1.0 - adx / 25)

        return RegimeEvent(
            asset             = self.symbol,
            regime            = regime,
            confidence        = round(confidence, 4),
            volatility_level  = vol,
            active_strategies = self.STRATEGY_MAP[regime],
            size_multiplier   = self.SIZE_MAP[regime],
            adx               = round(adx, 2),
            atr_ratio         = round(atr_ratio, 4),
            timestamp         = datetime.utcnow(),
        )

    def update(self, df: pd.DataFrame) -> RegimeEvent | None:
        """Returns a RegimeEvent only when a confirmed change occurs."""
        event = self.classify(df)
        self._buffer.append(event.regime)

        if len(self._buffer) == self.confirmation_bars:
            if len(set(self._buffer)) == 1:  # all bars agree
                candidate = self._buffer[-1]
                if candidate != self.current_regime:
                    self.current_regime = candidate
                    return event
        return None
```

```python
# messaging/bus.py
import redis, json
from datetime import datetime

r = redis.Redis(decode_responses=True)

CHANNELS = {
    "signals":         "signal_events",
    "regime_updates":  "regime_updates",
    "risk_events":     "risk_events",
    "execution":       "execution_reports",
    "system_halt":     "system_halt",
}

def publish(channel_key: str, payload: dict):
    payload["_ts"] = datetime.utcnow().isoformat()
    r.publish(CHANNELS[channel_key], json.dumps(payload))

def subscribe(channel_key: str):
    pubsub = r.pubsub()
    pubsub.subscribe(CHANNELS[channel_key])
    return pubsub
```

### 4. Agent listener pattern

Each agent subscribes to the regime channel and self-activates/deactivates:

```python
# agents/signal/base_agent.py
import threading
from messaging.bus import subscribe
import json

class BaseSignalAgent:
    def __init__(self, agent_id: str):
        self.agent_id    = agent_id
        self.active      = False
        self.size_mult   = 1.0
        self._start_regime_listener()

    def _start_regime_listener(self):
        def listen():
            pubsub = subscribe("regime_updates")
            for msg in pubsub.listen():
                if msg["type"] != "message":
                    continue
                event = json.loads(msg["data"])
                self.active    = self.agent_id in event.get("active_strategies", [])
                self.size_mult = event.get("size_multiplier", 1.0)
                status = "ACTIVE" if self.active else "INACTIVE"
                print(f"[{self.agent_id}] Regime: {event['regime']} → {status}")

        thread = threading.Thread(target=listen, daemon=True)
        thread.start()
```

### 5. Updated main loop

```python
# main.py  (v0.3.0)
import time, json
from data.fetcher import fetch_ohlcv
from agents.signal.trend_agent import TrendSignalAgent
from agents.signal.mean_reversion_agent import MeanReversionAgent
from agents.regime.regime_agent import RegimeAgent
from messaging.bus import publish
from brokers.alpaca_adapter import AlpacaAdapter
from risk.gate import BasicRiskGate
from risk.position_sizer import calculate_position_size
from logging.trade_log import log_trade
from config import ALPACA_API_KEY, ALPACA_SECRET_KEY, ALPACA_BASE_URL

SYMBOL   = "AAPL"
INTERVAL = 300   # 5 minutes

def run():
    broker  = AlpacaAdapter(ALPACA_API_KEY, ALPACA_SECRET_KEY, ALPACA_BASE_URL)
    equity  = broker.get_account_equity()
    gate    = BasicRiskGate(equity)
    regime  = RegimeAgent(SYMBOL)
    trend   = TrendSignalAgent(SYMBOL)
    mr      = MeanReversionAgent(SYMBOL)
    agents  = [trend, mr]

    while True:
        try:
            df = fetch_ohlcv(SYMBOL, period="60d", interval="1h")

            # 1. Check for regime change and broadcast
            regime_event = regime.update(df)
            if regime_event:
                publish("regime_updates", {
                    "regime":            regime_event.regime.value,
                    "active_strategies": regime_event.active_strategies,
                    "size_multiplier":   regime_event.size_multiplier,
                    "confidence":        regime_event.confidence,
                })
                print(f"[REGIME] → {regime_event.regime.value}")

            # 2. Collect signals from active agents
            for agent in agents:
                if not agent.active:
                    continue
                signal = agent.generate(df)
                if signal and signal.strength * signal.confidence >= 0.4:
                    publish("signals", {
                        "agent_id":  signal.agent_id,
                        "asset":     signal.asset,
                        "direction": signal.direction.value,
                        "strength":  signal.strength,
                        "confidence": signal.confidence,
                    })
                    print(f"[{signal.agent_id}] {signal.direction.value} "
                          f"str={signal.strength:.2f} conf={signal.confidence:.2f}")

        except Exception as e:
            print(f"⚠️  {e}")

        time.sleep(INTERVAL)

if __name__ == "__main__":
    run()
```
