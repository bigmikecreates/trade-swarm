```python
import yfinance as yf
import pandas as pd

def fetch_ohlcv(symbol: str, period: str = "2y", interval: str = "1h") -> pd.DataFrame:
    df = yf.download(symbol, period=period, interval=interval, progress=False)
    df.dropna(inplace=True)
    return df
```

**Step 3 — your first signal agent** (2–3 hours)

```python
# agents/signal/trend_agent.py
import pandas as pd
import pandas_ta as ta
from dataclasses import dataclass
from datetime import datetime
from enum import Enum

class Direction(Enum):
    LONG  = "long"
    SHORT = "short"
    FLAT  = "flat"

@dataclass
class SignalEvent:
    asset:      str
    direction:  Direction
    strength:   float       # 0.0 – 1.0
    confidence: float       # 0.0 – 1.0
    timestamp:  datetime
    indicators: dict

class TrendSignalAgent:
    def __init__(self, symbol: str):
        self.symbol = symbol

    def generate(self, df: pd.DataFrame) -> SignalEvent:
        df = df.copy()
        df.ta.ema(length=20,  append=True)
        df.ta.ema(length=50,  append=True)
        df.ta.adx(length=14,  append=True)
        df.ta.rsi(length=14,  append=True)
        df.dropna(inplace=True)

        last = df.iloc[-1]
        ema20  = last["EMA_20"]
        ema50  = last["EMA_50"]
        adx    = last["ADX_14"]
        rsi    = last["RSI_14"]

        # Direction
        if ema20 > ema50:
            direction = Direction.LONG
        elif ema20 < ema50:
            direction = Direction.SHORT
        else:
            direction = Direction.FLAT

        # Strength: how far apart are the EMAs?
        strength = min(abs(ema20 - ema50) / ema50 * 100, 1.0)

        # Confidence: ADX tells us if the trend is real
        confidence = min(adx / 50, 1.0)

        return SignalEvent(
            asset=self.symbol,
            direction=direction,
            strength=round(strength, 4),
            confidence=round(confidence, 4),
            timestamp=datetime.utcnow(),
            indicators={"ema20": ema20, "ema50": ema50, "adx": adx, "rsi": rsi}
        )
```

**Step 4 — backtest it** (1–2 hours)

```python
# backtest/run.py
import vectorbt as vbt
from data.fetcher import fetch_ohlcv
from agents.signal.trend_agent import TrendSignalAgent, Direction
import pandas_ta as ta

symbol = "EURUSD=X"   # start with one asset
df = fetch_ohlcv(symbol, period="2y", interval="1h")

# Compute signals for every bar
df.ta.ema(length=20, append=True)
df.ta.ema(length=50, append=True)
df.dropna(inplace=True)

entries = df["EMA_20"] > df["EMA_50"]
exits   = df["EMA_20"] < df["EMA_50"]

pf = vbt.Portfolio.from_signals(
    df["Close"],
    entries=entries,
    exits=exits,
    freq="1h",
    init_cash=10_000,
    fees=0.001       # 0.1% per trade
)

print(pf.stats())
# Key numbers to check: Sharpe Ratio, Max Drawdown, Win Rate, Total Return
```

## The single most important rule for v0.1.0

**Do not add more until the backtest passes your gate.** It's extremely tempting to start wiring Redis, adding agents, thinking about Docker. Resist all of it. You need a single strategy with a Sharpe above 0.8 and a max drawdown below 20% before anything else earns a place in the codebase.

Once that backtest is clean, move to v0.2.0 — paper trading via Alpaca. Everything after that builds on a proven foundation rather than hopeful code.
